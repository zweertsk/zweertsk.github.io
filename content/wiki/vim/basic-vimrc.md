```lua
syntax on
filetype plugin indent on

set relativenumber

set wildmenu
set wildignore=*.exe,*.dll,*.pdb,*/node_modules/*

set hidden

set spell
set is

set ignorecase
set smartcase

set cursorline
set ruler

set tabstop=4
set shiftwidth=4
set expandtab
```
